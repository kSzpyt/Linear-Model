Dobieranie zmiennych metod Hellwiga

Funkcja odpowiedzialna za metod Hellwiga

hellwig <- function(y, x)
{
  n <- ncol(x)
  l <- (2^n)-1
  R0 <- cor(y, x)
  R <- abs(as.matrix(cor(x)))
  argument <- replicate(n, c(0, 1), simplify = FALSE)
  comb <- as.matrix(expand.grid(argument))
  comb <- comb[-1,]
  h <- matrix(0, l, n)
  for(i in 1:l) 
  {
    for(j in 1:n)
    {
      h[i,j] <- (comb[i, j] * (R0[j]^2))/ (comb[i,] %*% as.vector(R[,j]))
    }
  }
  m=which.max(rowSums(h))
  colnames(comb) <- colnames(x)
  return((comb[m,]))
}

Dobieranie zmiennych metod bootstrapow

Funkcje odpowiedzialn za dobieranie zmiennych metod bootstrapow

boot_coef <- function(dane)
{
    boot_sample <- sample(1:nrow(dane), 
                          nrow(dane), 
                          replace = T)
    model <- lm(lp100km ~ ., data = dane[boot_sample, ])
    return(coef(model))
}
bootstrap <- function(dane, alpha = 0.05, n = N)
{
  coefs <- sapply(rep(1, n), function(x) boot_coef(dane))
  conf_inter <- apply(coefs, 1, function(x) quantile(x, c(alpha/2, 1-alpha/2)) )
  
  pvec<- data.frame("const" = NA, "cylinders" = NA, "displacement" = NA, "horsepower" = NA, "weight" = NA, "accelaration" = NA)
  
  coefs <- t(coefs)
  for(i in 1:ncol(conf_inter))
  {
    if(mean(conf_inter[1,i], conf_inter[2,i]) > 0)
    {
      pvec[, i] <- length(coefs[coefs[, i] <0 , i]) / n
    }
    else
    {
      pvec[, i] <- length(coefs[coefs[, i] > 0, i]) / n
    }
  }
  index <- 0
  index <- which(pvec[, -1] > alpha)
  if(length(index) != 0)
  {
      dane <- dane[, -(index + 1)]
      bootstrap(dane)
  }
  else
  {
    return(dane)
  }
}

##

P-value jest mniejsze od domylnego 5% co oznacza, e odrzucana jest hipoteza zerowa - w modelu wystpuje heteroskedastyczno.

Badanie poprawnoci specyfikacji dla modeli regresji liniowej

Test RESET Ramseya. Test stosowany jest w celu sprawdzenia, czy to liniowa posta modelu jest najlepszym moliwym do wybrania modelem.

resettest(model, power=2, type="regressor")
## 
##  RESET test
## 
## data:  model
## RESET = 1.7425, df1 = 2, df2 = 309, p-value = 0.1768

P-value jest wiksze od domylnego 5% co oznacza, e nie ma przesanek do odrzucenia H0 - posta modelu jest dobra.
